func PrintArray2D(arr: Array<Array<String>>, size1: Int, size2: Int):
{
	var i: Int = 0;
	while (i < size1)
	{
		var j: Int = 0;
		while (j < size2)
		{
			print("%s ", arr[i][j]);
			j = j + 1;
		}
		i = i + 1;
		print("\n");
	}
}

func main(argc: Int, argv: Array<String>):
{
	var arr: Array<Array<String>> = [
		["123", "345", "456"],
		["s"]
	];

	var subarr: Array<String> = arr[0];

	arr[1] = subarr;

	print("%s", arr[1][0]);
	print("%c", arr[1][0][2]);
}
