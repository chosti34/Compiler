func PrintArray2D(arr: Array<Array<String>>, size1: Int, size2: Int):
{
	var i: Int = 0;
	while (i < size1)
	{
		var j: Int = 0;
		while (j < size2)
		{
			print("%s ", arr[i][j]);
			j = j + 1;
		}
		i = i + 1;
		print("\n");
	}
}

func main(argc: Int, argv: Array<String>):
{
	var arr: Array<Array<String>> = [["123", "345", "456"], ["s"]];
	
	arr[0][1] = "asd";
	
	print("%s\n", arr[0][0]);
	print("%s\n", arr[0][1]);
	print("%s\n", arr[0][2]);
	print("%s\n", arr[1][0]);
}
