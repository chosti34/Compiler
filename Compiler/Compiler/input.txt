func sum(array: Array<Int>, size: Int) -> Int:
{
	if (size == 0)
	{
		return -1;
	}

	var sum: Int = array[0];
	var index: Int = 1;

	while (index < size)
	{
		sum = sum + array[index];
		index = index + 1;
	}

	return sum;
}
